## Programmation du template "profil" :

if (id in data.path) is data.me.id then
	data.relation = "moi"
else
	if id is in data.me.pairs then data.relation = "pair"
	if id is in data.me.groupes then data.relation = "groupe"
	else data.relation = "membre"
end if
