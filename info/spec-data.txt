## Programmation du template "profil"
if (id in data.path) is membre.id then
	data.relation = "moi"
else
	if id is in membre.pairs then data.relation = "pair"
	if id is in membre.groupes then data.relation = "groupe"
	else data.relation = "membre"
end if
